Трекер расходов — настройка за 5 минут
1

Создай Google Таблицу

1 мин

Открой sheets.new → назови таблицу как хочешь.
В строку 1 впиши заголовки:

Дата Описание Категория Сумма

Каждое слово в отдельную ячейку: A1, B1, C1, D1.

2

Скопируй ID таблицы

30 сек

Из адресной строки скопируй ID — длинная строка между /d/ и /edit:

https://docs.google.com/spreadsheets/d/ВОТ_ЭТОТ_ID/edit
3

Добавь скрипт

2 мин

В таблице: Расширения → Apps Script
Удали всё что там есть. Вставь этот код:

var SHEET_ID = "ВСТАВЬ_СВОЙ_ID"; function getSheet() { var ss = SpreadsheetApp.openById(SHEET_ID); return ss.getSheetByName("Расходы") || ss.getSheets()[0]; } function doGet(e) { var p = e.parameter; if (p.action === "add") { var sheet = getSheet(); sheet.appendRow([p.date, p.description, p.category, Number(p.amount)]); return ContentService.createTextOutput("ok"); } return ContentService.createTextOutput("Трекер работает! Записей: " + Math.max(0, getSheet().getLastRow() - 1)); } function doPost(e) { try { var data = JSON.parse(e.parameter.payload || e.postData.contents); var sheet = getSheet(); if (data.action === "sync_all") { var lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.getRange(2, 1, lastRow - 1, 4).clear(); var rows = data.expenses.map(function(exp) { return [exp.date, exp.description, exp.category, exp.amount]; }); if (rows.length > 0) sheet.getRange(2, 1, rows.length, 4).setValues(rows); return ContentService.createTextOutput("ok:" + rows.length); } return ContentService.createTextOutput("error:unknown_action"); } catch (err) { return ContentService.createTextOutput("error:" + err.toString()); } }

⚠️ В первой строке замени ВСТАВЬ_СВОЙ_ID на ID из шага 2.

Разверни скрипт

1 мин

В Apps Script:

1. Нажми Развертывание → Новое развертывание
2. Слева нажми ⚙️ → Веб-приложение
3. Выполнять от: Меня
4. Доступ: Все
5. Нажми Развернуть
6. Скопируй URL

Проверка: открой URL в браузере — должно написать "Трекер работает!"

} var pieChart=d.newChart().setChartType(Charts.ChartType.PIE) .addRange(d.getRange(catDataRow,2,15,2)).setPosition(catStart,6,0,0) .setOption("title","Расходы по категориям").setOption("titleTextStyle",{color:txtL,fontSize:12}) .setOption("legend",{position:"right",textStyle:{color:txtL,fontSize:10}}) .setOption("backgroundColor",bgDark).setOption("pieHole",0.4) .setOption("colors",["#F97316","#3B82F6","#8B5CF6","#10B981","#EC4899","#F59E0B","#06B6D4","#6366F1","#F43F5E","#84CC16","#94A3B8"]) .setOption("chartArea",{width:"90%",height:"80%"}).setOption("width",500).setOption("height",300).build(); Затем выбери функцию createDashboard → нажми ▶ Run.
Появится лист «Аналитика» с графиками. Данные обновляются автоматически при каждом новом расходе.

💡 Расходы сохраняются в Google Sheets навсегда. Локально на телефоне тоже хранятся для быстрой работы, но если пропадут — данные в таблице целы.
